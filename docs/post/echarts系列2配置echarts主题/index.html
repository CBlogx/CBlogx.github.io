// 引入主题
import theme from "theme.json";

// 使用echarts
import echarts from "echarts";
echarts.registerTheme("default", theme);

// 使用vue-echarts
import echarts from "vue-echarts";
import { registerTheme } from "echarts/core";
registerTheme("default", theme);
